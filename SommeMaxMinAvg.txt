 //4.1 Reducer_Somme
 Algorithme Reducer_Somme
 Variables
    sum : Entier
    key : Texte
    values : Itérateur d'Entier
 Début
    Procédure reduce(key, values, output, reporter)
    Début
        sum ← 0
        TantQue values a un élément Faire
            sum ← sum + values.Suivant()
        FinTantQue
        output.Collecte(key, sum)
    FinProcédure
 FinAlgorithme

 //4.4 Reducer_Moyenne
 Algorithme Reducer_Moyenne
 Variables
    sum : Entier
    count : Entier
    moyenne : Réel
    key : Texte
    values : Itérateur d'Entier
 Début
    Procédure reduce(key, values, output, reporter)
    Début
        sum ← 0
        count ← 0
        TantQue values a un élément Faire
            val ← values.Suivant()
            sum ← sum + val
            count ← count + 1
        FinTantQue
moyenne ← sum / count
 output.Collecte(key, moyenne)
 FinProcédure
 FinAlgorithme
